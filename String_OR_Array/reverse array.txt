#include <iostream>

using namespace std;

int main()
{
    int n,a[10000];
   cin >> n;
   for(int i=0;i<n;i++){
       cin >> a[i];
   }
   int i,j;
   for(i=0,j=n-1;i<n/2;i++,j--)
   {
       int t = a[i];
       a[i] = a[j];
       a[j] = t;
   }
   for(int i=0;i<n;i++){
       cout << a[i] << " ";
    }

    return 0;
}
